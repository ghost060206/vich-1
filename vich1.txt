#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

// Решение СЛАУ методом релаксации
void relax(const vector<vector<double>>& A, const vector<double>& b,
           double omega, double eps, int maxIter, vector<double>& x) {
    int n = A.size();
    x.assign(n, 0.0);               // начальное приближение – нули
    vector<double> x_old(n, 0.0);

    for (int iter = 0; iter < maxIter; ++iter) {
        for (int i = 0; i < n; ++i) {
            double sum = 0.0;
            for (int j = 0; j < i; ++j)      // уже обновлённые компоненты
                sum += A[i][j] * x[j];
            for (int j = i + 1; j < n; ++j)  // старые компоненты
                sum += A[i][j] * x_old[j];

            x[i] = (1.0 - omega) * x_old[i] + (omega / A[i][i]) * (b[i] - sum);
        }

        double diff = 0.0;
        for (int i = 0; i < n; ++i) {
            diff += (x[i] - x_old[i]) * (x[i] - x_old[i]);
            x_old[i] = x[i];
        }
        if (sqrt(diff) < eps) return;   // достигнута требуемая точность
    }
}

int main() {
    int n;
    cout << "Размер системы: "; cin >> n;

    vector<vector<double>> A(n, vector<double>(n));
    vector<double> b(n);

    cout << "Матрица A:\n";
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            cin >> A[i][j];

    cout << "Вектор b:\n";
    for (int i = 0; i < n; ++i) cin >> b[i];

    double omega, eps;
    int maxIter;
    cout << "omega (0<ω<2), eps, maxIter: ";
    cin >> omega >> eps >> maxIter;

    vector<double> x;
    relax(A, b, omega, eps, maxIter, x);

    cout << "Решение:\n";
    for (double v : x) cout << v << " ";
    cout << endl;

    return 0;
}
